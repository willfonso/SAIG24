Configuración de Google Sheets

Sigue estos pasos para vincular los resultados de la encuesta a tu hoja de cálculo:

  1. Crea una nueva hoja de cálculo en Google Sheets.
  2. Ve a Extensiones > Apps Script en el menú superior.
  3. Borra el código existente y pega el siguiente script: 
    function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "Fecha", 
      "ID Sucursal",
      "Nombre Sucursal",
      "Experiencia de Compra", 
      "Encontró lo que Buscaba", 
      "Calificación Atención", 
      "Limpieza y Orden", 
      "Recomendaría a Amigos", 
      "Mejoras Sugeridas"
    ]);
  }
  
  try {
    var data = JSON.parse(e.postData.contents);
    sheet.appendRow([
      new Date(),
      data.sucursalId,
      data.sucursalNombre,
      data.experiencia,
      data.encontrado,
      data.atencion,
      data.limpieza,
      data.recomendar,
      data.mejoras
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({ "status": "success" }))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() }))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  }
}

function doOptions(e) {
  return ContentService.createTextOutput("")
    .setHeader("Access-Control-Allow-Origin", "*")
    .setHeader("Access-Control-Allow-Methods", "POST, GET, OPTIONS")
    .setHeader("Access-Control-Allow-Headers", "Content-Type");
}
  4. Haz clic en el botón de Guardar (icono de disco).
  5. Haz clic en Implementar > Nueva implementación.
  6. Selecciona el tipo de implementación haciendo clic en el engranaje y eligiendo Aplicación web.
  7. Configura los campos:
    • Descripción: Webhook Encuesta SAIG24
    • Ejecutar como: Yo (tu cuenta)
    • Quién tiene acceso: Cualquier persona (esto permite que la web envíe datos)
  8. Haz clic en Implementar. Autoriza los accesos si Google lo solicita.
  9. Copia la URL de la aplicación web generada.
